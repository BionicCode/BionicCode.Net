    Class BoolToStringConverter

    Implementation of System.Windows.Data.IValueConverter that converts a System.Boolean to a custom string representation e.g., convert true to "Enabled".

    Examples
    <ToggleButton IsChecked="{Binding IsEnabled}">
  <ToggleButton.Content>
    <Binding Path="IsEnabled">
      <Binding.Converter>
        <BoolToStringConverter TrueValue="On" FalseValue="{Binding DisabledText}" NullValue="Undefined" />
      </Binding.Converter>
    </Binding>
  </ToggleButton.Content>
</ToggleButton>
